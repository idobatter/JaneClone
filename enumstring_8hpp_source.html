136 template <class D, class E>
137 const std::string &EnumStringBase<D,E>::From(const E e)
138 {
139  for(;;) // Code block
140  {
141  // Search for the enumerator in our map
142  typename AssocMap::const_iterator i;
143  for(i = GetMap().begin(); i != GetMap().end(); ++i)
144  if( (*i).second == e )
145  break;
146 
147  // If we didn't find it, we can't do this conversion
148  if( i == GetMap().end() )
149  break;
150 
151  // Keep searching and see if we find another one with the same value
152  typename AssocMap::const_iterator j( i );
153  for(++j; j != GetMap().end(); ++j)
154  if( (*j).second == e )
155  break;
156 
157  // If we found another one with the same value, we can't do this conversion
158  if( j != GetMap().end() )
159  break;
160 
161  // We found exactly one string which matches the required enumerator
162  return (*i).first;
163  }
164 
165  // We couldn't do this conversion; return an empty string.
166  static const std::string dummy;
167  return dummy;
168 }

170 template <class D, class E>
171 const bool EnumStringBase<D,E>::To(E &e, const std::string &str)
172 {
173  // Search for the string in our map.
174  const typename AssocMap::const_iterator itr( GetMap().find( str ) );
175 
176  // If we have it, then return the associated enumerator.
177  if( itr != GetMap().end() )
178  {
179  e = (*itr).second;
180  return true;
181  }
182 
183  // We don't have it; the conversion failed.
184  return false;
